# How to run frontend
    cd client
    npm install
    npm start


# How to run backend
    cd backend
    npm install
    npm start

# how to run standalone worker
    cd worker
    npm install
    npm start

# how to run redis and  mongo if you have docker
    docker run -p 27017:27017 -idt mongo
    docker run -p 6379:6379 -idt redis
